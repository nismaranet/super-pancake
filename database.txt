CREATE TABLE public.profiles (
  id uuid NOT NULL,
  username text UNIQUE CHECK (char_length(username) = 3 AND char_length(username) = 20 AND username !~ 's'text),
  avatar_url text,
  steam_guid text UNIQUE CHECK (steam_guid IS NULL OR steam_guid ~ '^[0-9]{17}$'text),
  discord_id text UNIQUE,
  updated_at timestamp with time zone DEFAULT timezone('utc'text, now()),
  role text DEFAULT 'user'text CHECK (role = ANY (ARRAY['user'text, 'admin'text, 'moderator'text])),
  safety_rating double precision DEFAULT 2.50,
  total_xp integer DEFAULT 0,
  driver_level integer DEFAULT 1,
  display_name text UNIQUE CHECK (char_length(display_name) = 3 AND char_length(display_name) = 30),
  total_starts integer DEFAULT 0,
  total_wins integer DEFAULT 0,
  total_podiums integer DEFAULT 0,
  unranked_wins integer DEFAULT 0,
  unranked_podiums integer DEFAULT 0,
  unranked_starts integer DEFAULT 0,
  total_distance_km double precision DEFAULT 0.0,
  CONSTRAINT profiles_pkey PRIMARY KEY (id),
  CONSTRAINT profiles_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
